/*20. For every database, rank them according to number of indexes have created*/

select schema_name, count(i.attributes_id) as number_idx, RANK() OVER (ORDER BY count(i.attributes_id) desc) as rank
from DG_Schema S join DG_Tables T 
on S.schema_id = T.database_id
join DG_Attributes A on A.table_id = T.table_id
join DG_Attributes_Index I on I.attributes_id = A.attributes_id
group by s.schema_name;

---------- ---------- ----------
Zeota             141          1
Prop               19          2
SP                  2          3
